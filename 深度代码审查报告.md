#### 1. 数据同步一致性问题
**问题描述**：
- 教师端查看学生进度时，SQLite读取存在延迟
- 多用户演示时，数据无法实时更新
- 缺少数据刷新机制

**影响**：

- 多人同时操作时数据不一致

**修复方案**：
```javascript
// backend/src/api/teacherRoutes.js
// 添加数据缓存和实时更新机制

const dataCache = new Map();
const CACHE_TTL = 5000; // 5秒缓存

// 包装数据库查询，添加缓存和错误处理
function cachedQuery(queryFn, cacheKey) {
  return new Promise((resolve, reject) => {
    // 检查缓存
    const cached = dataCache.get(cacheKey);
    if (cached && Date.now() - cached.timestamp < CACHE_TTL) {
      resolve(cached.data);
      return;
    }
    
    // 执行查询
    queryFn().then(data => {
      // 更新缓存
      dataCache.set(cacheKey, {
        data,
        timestamp: Date.now()
      });
      resolve(data);
    }).catch(reject);
  });
}

// 修改dashboard接口，添加实时数据推送
router.get('/dashboard', authenticateTeacher, async (req, res) => {

        db.all('SELECT * FROM v_student_learning_stats ORDER BY poem_count DESC', (err, rows) => {
```

```javascript
// 添加数据库操作重试机制

const MAX_RETRIES = 3;
const RETRY_DELAY = 100;

function runWithRetry(sql, params, retries = MAX_RETRIES) {
  return new Promise((resolve, reject) => {
    const attempt = (attemptNumber) => {
      db.run(sql, params, function(err) {
        if (err) {
          if (err.code === 'SQLITE_BUSY' && attemptNumber < retries) {
            console.log(`数据库锁定，重试 ${attemptNumber}/${retries}`);
            setTimeout(() => attempt(attemptNumber + 1), RETRY_DELAY * attemptNumber);
          } else {
            reject(err);
          }
        } else {
          resolve({ id: this.lastID, changes: this.changes });
        }
      });
    };
    attempt(1);
  });
}
```

## 任务二：飞花令联机 (Socket Online Battle) 逻辑审查

### 【严重风险】

#### 1. 并发竞态问题
**问题描述**：
- 两名玩家几乎同时发送答案时，可能同时通过轮次检查
- 存在双赢或双输的逻辑漏洞
- 缺少原子性操作保证

**修复方案**：
```javascript
// backend/src/socket/index.js
// 添加并发控制和原子性操作

const roomLocks = new Map(); // 房间锁

// 获取房间锁
async function acquireRoomLock(roomId) {
  while (roomLocks.get(roomId)) {
    await new Promise(resolve => setTimeout(resolve, 10));
  }
  roomLocks.set(roomId, true);
}

// 释放房间锁
function releaseRoomLock(roomId) {
  roomLocks.delete(roomId);
}

// 提交诗句（添加并发控制）
socket.on('submit-poem', async (data) => {
  const { roomId, poem } = data;
  
  try {
    // 获取房间锁
    await acquireRoomLock(roomId);
    
    const room = gameRooms.get(roomId);
    
    if (!room) {
      releaseRoomLock(roomId);
      socket.emit('error', { message: '房间不存在' });
      return;
    }
    
    // 双重检查：验证是否轮到当前用户
    const currentPlayerIndex = room.players.findIndex(p => p.userId === socket.userId);
    
    if (currentPlayerIndex === -1) {
      releaseRoomLock(roomId);
      socket.emit('error', { message: '你不在该房间中' });
      return;
    }
    
    if (currentPlayerIndex !== room.currentTurn) {
      releaseRoomLock(roomId);
      socket.emit('error', { message: '还没轮到你' });
      return;
    }
    
    // 验证诗句
    if (!poem || typeof poem !== 'string') {
      releaseRoomLock(roomId);
      socket.emit('error', { message: '诗句格式错误' });
      return;
    }
    
    // 检查是否包含令字（严格验证）
    const keyword = room.keyword;
    if (!poem.includes(keyword)) {
      releaseRoomLock(roomId);
      socket.emit('error', { message: '诗句不包含令字' });
      return;
    }
    
    // 检查是否重复（使用Set提高性能）
    if (room.usedPoemsSet && room.usedPoemsSet.has(poem)) {
      releaseRoomLock(roomId);
      socket.emit('error', { message: '该诗句已被使用' });
      return;
    }
    
    // 原子性更新房间状态
    room.usedPoems = room.usedPoems || [];
    room.usedPoemsSet = room.usedPoemsSet || new Set();
    
    room.usedPoems.push(poem);
    room.usedPoemsSet.add(poem);
    room.players[currentPlayerIndex].score += 1;
    room.currentTurn = (room.currentTurn + 1) % 2;
    room.currentRound += 1;
    
    // 广播给房间内所有玩家
    io.to(roomId).emit('poem-submitted', {
      poem,
      playerId: socket.userId,
      playerName: room.players[currentPlayerIndex].username,
      currentTurn: room.currentTurn,
      currentRound: room.currentRound,
      usedPoems: room.usedPoems,
      players: room.players
    });
    
    console.log(`玩家 ${socket.userId} 在房间 ${roomId} 提交了诗句: ${poem}`);
    
    // 释放房间锁
    releaseRoomLock(roomId);
    
  } catch (error) {
    console.error('提交诗句错误:', error);
    releaseRoomLock(roomId);
    socket.emit('error', { message: '提交失败，请重试' });
  }
});
```

#### 2. 断线重连处理
**问题描述**：
- 断线后游戏状态卡死
- 缺少掉线判定和自动结算
- 没有快速重连机制

**修复方案**：
```javascript
// backend/src/socket/index.js
// 增强断线处理和重连机制

const disconnectedPlayers = new Map(); // 断线玩家记录

// 断开连接处理
socket.on('disconnect', async () => {
  console.log('用户断开连接:', socket.id);
  
  // 记录断线时间
  const disconnectTime = Date.now();
  
  // 从在线列表中移除
  for (const [userId, user] of onlineUsers.entries()) {
    if (user.socketId === socket.id) {
      onlineUsers.delete(userId);
      console.log('用户离线:', user.username);
      
      // 记录断线玩家，给予30秒重连时间
      disconnectedPlayers.set(userId, {
        userId,
        username: user.username,
        disconnectTime,
        socketId: socket.id
      });
      
      break;
    }
  }
  
  // 广播在线用户列表更新
  await broadcastOnlineUsers(io);
  
  // 处理游戏中的断线
  for (const [roomId, room] of gameRooms.entries()) {
    if (room.players.some(p => p.userId === socket.userId)) {
      const otherPlayer = room.players.find(p => p.userId !== socket.userId);
      
      // 给予30秒重连等待
      setTimeout(async () => {
        // 检查是否重连
        const disconnected = disconnectedPlayers.get(socket.userId);
        if (disconnected && Date.now() - disconnected.disconnectTime >= 30000) {
          // 超时未重连，判定断线方失败
          await saveBattleRecord(room, otherPlayer.userId, socket.userId);
          
          io.to(roomId).emit('opponent-disconnected', {
            winnerId: otherPlayer.userId,
            winnerName: otherPlayer.username,
            reason: 'disconnect_timeout'
          });
          
          const otherUser = onlineUsers.get(otherPlayer.userId);
          if (otherUser) otherUser.inGame = false;
          
          gameRooms.delete(roomId);
          disconnectedPlayers.delete(socket.userId);
        }
      }, 30000);
      
      // 通知对方玩家断线
      io.to(roomId).emit('opponent-disconnecting', {
        playerId: socket.userId,
        message: '对方已断线，等待重连...',
        timeout: 30
      });
    }
  }
});

// 重连处理
socket.on('reconnect-game', async (data) => {
  const { roomId, userId } = data;
  
  const room = gameRooms.get(roomId);
  if (!room) {
    socket.emit('error', { message: '游戏已结束' });
    return;
  }
  
  const player = room.players.find(p => p.userId === userId);
  if (!player) {
    socket.emit('error', { message: '你不在该游戏中' });
    return;
  }
  
  // 检查断线记录
  const disconnected = disconnectedPlayers.get(userId);
  if (!disconnected) {
    socket.emit('error', { message: '重连超时' });
    return;
  }
  
  // 恢复游戏
  disconnectedPlayers.delete(userId);
  
  // 更新socket信息
  player.socketId = socket.id;
  socket.join(roomId);
  
  // 发送游戏状态
  socket.emit('game-reconnected', {
    room: gameRooms.get(roomId),
    message: '重连成功'
  });
  
  // 通知对方
  socket.to(roomId).emit('opponent-reconnected', {
    playerId: userId,
    message: '对方已重连'
  });
});
```

```javascript
// 服务器端计时器
function startServerTimer(roomId, io) {
  const room = gameRooms.get(roomId);
  if (!room) return;
  
  // 清除旧计时器
  if (roomTimers.has(roomId)) {
    clearInterval(roomTimers.get(roomId));
  }
  
  // 启动新计时器
  const timer = setInterval(() => {
    const currentRoom = gameRooms.get(roomId);
    if (!currentRoom) {
      clearInterval(timer);
      roomTimers.delete(roomId);
      return;
    }
    
    const elapsed = Date.now() - currentRoom.turnStartTime;
    const remaining = Math.max(0, currentRoom.timeLimit * 1000 - elapsed);
    
    // 广播剩余时间
    io.to(roomId).emit('time-update', {
      remaining: Math.ceil(remaining / 1000),
      serverTime: Date.now()
    });
    
    // 超时处理
    if (remaining <= 0) {
      clearInterval(timer);
      roomTimers.delete(roomId);
      
      // 判定当前玩家超时失败
      const loserIndex = currentRoom.currentTurn;
      const winnerIndex = (loserIndex + 1) % 2;
      
      handleTimeout(roomId, currentRoom.players[winnerIndex].userId, 
                    currentRoom.players[loserIndex].userId, io);
    }
  }, 1000); // 每秒更新一次
  
  roomTimers.set(roomId, timer);
}

// 处理超时
async function handleTimeout(roomId, winnerId, loserId, io) {
  const room = gameRooms.get(roomId);
  if (!room) return;
  
  await saveBattleRecord(room, winnerId, loserId);
  
  io.to(roomId).emit('game-result', {
    winnerId,
    winnerName: room.players.find(p => p.userId === winnerId).username,
    loserId,
    loserName: room.players.find(p => p.userId === loserId).username,
    totalRounds: room.currentRound,
    reason: 'timeout'
  });
  
  // 清理
  gameRooms.delete(roomId);
  
  const winnerUser = onlineUsers.get(winnerId);
  const loserUser = onlineUsers.get(loserId);
  if (winnerUser) winnerUser.inGame = false;
  if (loserUser) loserUser.inGame = false;
  
  await broadcastOnlineUsers(io);
}
```
